###Code example:
Create a program to find the sum of the minimum and maximum of three entered numbers.

```
function calcSum (a, b, c){
let max
    if(a > b && a > c){
        max = a
    }
    else if (b > a && b > c){
        max = b
    }
    else max = c

let min
    if(a < b && a < c){
        min = a
    }
    else if (b < a && b < c){
        min = b
    }
    else min = c

    return max + min
}
```
